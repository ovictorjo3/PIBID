\section{DefiniĂ§ĂŁo}

	\begin{frame}{DefiniĂ§ĂŁo}
		Sendo $a$ e $b$ nĂșmeros reais e positivos, com $a \neq 1$, chama-se \textbf{logaritmo de b na base a} o expoente $x$ ao qual se deve elevar a base $a$ de modo que a potĂȘncia $a^{x}$ seja igual a $b$.
		
		$$\log_{a}b=x \Leftrightarrow a^{x}=b$$
		
		\uncover<2->{
		\begin{block}{Dizemos que:}
			\begin{itemize}
				\item $a$ Ă© a \textbf{base} do logaritmo;
				\item $b$ Ă© o \textbf{logaritmando};
				\item $x$ Ă© o \textbf{logaritmo}.
			\end{itemize}
		\end{block}
		}
	\end{frame}

	\begin{frame}{DefiniĂ§ĂŁo}
		
		\begin{figure}
			\centering
			\includegraphics[scale=0.5]{imagens/ex-poliedro.png}
			\caption{Sistema de Ensino Poliedro}
		\end{figure}
	
	\end{frame}

    \begin{frame}{Exemplos}
       
       Vejamos alguns exemplos de logaritmos:
       
       \begin{itemize}
       		\item [a)] $\log_{2}(8)=3, \ \text{pois} \ 2^{3}=8$ \\ 
       		\uncover<2->{\item [b)] $\log_{3}9=2, \ \text{pois} \ 3^{2}=9 $} \\
       		\uncover<3->{\item [c)] $\log_{4}1=0, \ \text{pois} \ 4^{0}=1 $} \\
       		\uncover<4->{\item [d)] $\log_{\frac{1}{2}}8=-3, \ \text{pois} \ \left( \dfrac{1}{2} \right)^{-3}=8 $} \\
       		\uncover<5->{\item [e)] $\log_{0,5}0,25=2, \ \text{pois} \ \left(0,5 \right)^{2}=0,25 $}
       	\end{itemize}
    \end{frame}

    \begin{frame}{ConvenĂ§ĂŁo importante}
    	
    	\begin{block}{Logaritmo decimal}
    		\justifying
    		Quando a base do logaritmo for 10, podemos omiti-la da notaĂ§ĂŁo:
    		\begin{itemize}
    			\item $\log1 = \log_{10}1=0$
    			\item $\log1000=\log_{10}1000=3$
    		\end{itemize}
    	\end{block}
    	
    	\uncover<2->{
    	\begin{block}{Logaritmo natural (Logaritmo neperiano)}
    		\justifying
    		JĂĄ quando a base do logaritmo Ă© o nĂșmero de Euler, $\euler \approx 2,7183$, a notaĂ§ĂŁo utilizada Ă© $\ln$:
    		\begin{itemize}
    			\item $\ln e = \log_{\euler}\euler = 1$
    			\item $\ln \euler^{3}=\log_{\euler}\euler^{3}=3$
    		\end{itemize}
    	\end{block}
    	}
    	
    \end{frame}

\section{ConsequĂȘncias}
	
	\begin{frame}{ConsequĂȘncias}
		Sejam $a,b$ e $c$ nĂșmeros reais com $0<a\neq 1, b>0 \ \text{e} \ c>0$. Decorrem da definiĂ§ĂŁo de logaritmo as seguintes propriedades (P):
  
	
		\begin{block}{P1: O logaritmo de 1 em qualquer base $a$ Ă© igual a $0$.}
			$$\log_{a}1=0,\  \text{pois} \ a^{0}=1$$
		\end{block}
		
		\uncover<2->{\begin{block}{P2: O logaritmo da base, qualquer que seja ela, Ă© igual a 1.}
			$$\log_{a}a=1,\  \text{pois} \ a^{1}=a$$
		\end{block}}
		
		\uncover<3->{\begin{block}{P3: A potĂȘncia de base $a$ e expoente $\log_{a}b$ Ă© igual a b.}
			$$a^{\log_{a}b}=b$$
		\end{block}}
	
	\end{frame}
	
	\begin{frame}{ConsequĂȘncias}
		Sejam $a,b$ e $c$ nĂșmeros reais com $0<a\neq 1, b>0 \ \text{e} \ c>0$. Decorrem da definiĂ§ĂŁo de logaritmo as seguintes propriedades (P):
		
		
		\begin{block}{P4:}
			\begin{itemize}
				\justifying
				\item Se dois logaritmos em uma mesma base sĂŁo iguais, entĂŁo os logaritmandos tambĂ©m sĂŁo iguais. $$ \log_{a}b=\log_{a}c \Rightarrow  b=c$$
			
				\item Reciprocamente, se dois nĂșmeros reais positivos sĂŁo iguais, seus logaritmos em uma mesma base tambĂ©m sĂŁo iguais.
				
			\end{itemize}
		\end{block}
	\end{frame}
 
	\begin{frame}{AplicaĂ§ĂŁo}
		\begin{alertblock}{\exercicio}
			Qual Ă© o valor de $9^{\log_{3}5}$?
		\end{alertblock}

		\end{alertblock}
		
		
		\begin{alertblock}{\exercicio}

	\end{frame}
	
\section{Propriedades operatĂłrias}
	
	\begin{frame}{Propriedades operatĂłrias (PO)}
		Vamos  agora estudar trĂȘs propriedades operatĂłrias envolvendo logaritmos. \\
		
		\begin{block}{PO1: Logaritmo do produto}
			\justifying
			Em qualquer base, o logaritmo do produto de dois nĂșmeros reais e positivos Ă© igual Ă  soma dos logaritmos de cada um deles, isto Ă©, se $0<a\neq 1, \ b>0 \ \text{e} \ c>0$, entĂŁo: $$\log_{a}(b.c)=\log_{a}b+\log_{a}c$$
		\end{block} 
		
		Por exemplo: $$\log_{2}6=\log_{2}(2.3)=\log_{2}2+\log_{2}3=1+\log_{2}3$$
	\end{frame}
	
	\begin{frame}{Propriedades operatĂłrias (PO)}
		
		
		\begin{block}{PO2: Logaritmo do quociente}
			\justifying
			Em qualquer base, o logaritmo do quociente de dois nĂșmeros reais e positivos Ă© igual Ă  diferenĂ§a entre o logaritmo do numerador e o logaritmo do denominador, isto Ă©, se $0<a\neq 1, \ b>0 \ \text{e} \ c>0$, entĂŁo: $$\log_{a}\left( \frac{b}{c} \right)=\log_{a}b-\log_{a}c$$
		\end{block} 
		
		Por exemplo: $$\log\left(\frac{3}{100} \right)=\log3-\log100=\log3-2$$
	\end{frame}
	
	\begin{frame}{Propriedades operatĂłrias (PO)}
		
		
		\begin{block}{PO3: Logaritmo da potĂȘncia}
			\justifying
			Em qualquer base, o logaritmo de uma potĂȘncia de base real e positiva Ă© igual ao produto do expoente pelo logaritmo da base da potĂȘncia, isto Ă©, se $0<a\neq 1, \ b>0 \ \text{e} \ r \in \mathbb{R}$, entĂŁo: $$\log_{a}b^{r}=r\log_{a}b$$
		\end{block} 
		
		Por exemplo: $$\log_{5}27=\log_{5}3^{3}=3\log_{5}3$$
	\end{frame}
	
	
	\begin{frame}{AplicaĂ§ĂŁo}
		\begin{alertblock}{\exercicio}
			\justifying
			Calcular o valor de $\log_{b}\left(x^{2}.y\right)$, sabendo que $\log_{b}x=3$ e $\log_{b}y=-4 \left(x>0, y>0 \ \text{e} \ 0<b\neq 1 \right)$
		\end{alertblock}
		
		\begin{alertblock}{\exercicio}
			\justifying
			Qual Ă© a expressĂŁo E cujo desenvolvimento logarĂ­tmico (em base 10) Ă© $\log\text{E} = 1+\log\text{a}+2\log\text{b}-\log\text{c}$, com a, b e c nĂșmeros reais e positivos?
		\end{alertblock}
	
	
		\begin{alertblock}{\exercicio}
			\justifying
			Admitindo que $\log2=\text{a}$ e $\log3=\text{b}$, obter o valor de $\log0,48$, em funĂ§ĂŁo de $a$ e $b$.
		\end{alertblock}
	
	\end{frame}

\section{MudanĂ§a de base}

	\begin{frame}{MudanĂ§a de base}
		\begin{block}{DefiniĂ§ĂŁo}
				Suponha que $a, b \ \text{e} \ c$ nĂșmeros reais positivos, com $a \ \text{e} \ b$ diferentes de $1$. Temos:
			
				$$\log_{a}c=\dfrac{\log_{b}c}{\log_{b}a}$$
		\end{block} Por exemplo:
		
		\begin{itemize}
			
			\uncover<2->{
			\item [a)] base 10: $\log_{2}5=\dfrac{\log_{10}5}{\log_{10}2}=\dfrac{0,699}{0,3010} \approx 2,32$ \\
			}
			
			\vspace{0.5cm}
			
			\uncover<3->{
			\item [b)] base \euler: $\log_{2}5=\dfrac{\log_{\euler}5}{\log_{\euler}2}=\dfrac{\ln 5}{\ln 2}=\dfrac{1,609}{0,693} \approx 2,32$
			}
		\end{itemize}
	
	\end{frame}
	
	\begin{frame}{ExercĂ­cio}
		\begin{alertblock}{\exercicio}
			\justifying
			Calcular o valor de $\log_{100}72$, considerando as aproximaĂ§Ă”es: $\log2 = 0,3 \ \text{e} \ \log3=0,48$.
		\end{alertblock}
	\end{frame}
	
	
	\begin{frame}{AplicaĂ§ĂŁo importante}
		Sejam $a \ \text{e} \ b$ reais positivos e diferentes de $1$. Temos que:
		
		\begin{block}{DefiniĂ§ĂŁo}
			$$\log_{a}b \ . \ \log_{b}a=1 \ \text{ou} \ \log_{b}a=\dfrac{1}{\log_{a}b}$$
		\end{block} Por exemplo:
		
		\begin{itemize}
			\uncover<2->{
			\item [a)] $\log_{3}2\ . \ \log_{2}3=1$ \\
			}
			
			\vspace{0.3cm}
			
			\uncover<3->{
			\item [b)] $\log_{4}5=\dfrac{1}{\log_{5}4}$
			}
		\end{itemize}
		
	\end{frame}
